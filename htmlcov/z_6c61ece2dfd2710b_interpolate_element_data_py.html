21class CalculationType(Enum): 

22 """Enum representing different calculation types for element value 

23 determination. 

24 

25 This enum defines the available calculation types for mapping image data 

26 to finite element Discretization elements. The calculation type determines 

27 how the image pixel data is assigned to the elements in the 

28 Discretization. 

29 

30 Attributes: 

31 NODES (str): Represents the calculation method where the pixel value 

32 is averaged over the nodes of the element. 

33 CENTER (str): Represents the calculation method where the pixel value 

34 is based on the center of the element. 

35 ALLVOXELS (str): Represents the calculation method where the pixel 

36 value is averaged over all voxels inside the element. 

37 """ 

38 

39 NODES = "nodes" 

40 CENTER = "elementcenter" 

41 ALLVOXELS = "allvoxels"

43 def get_interpolator(self) -> Type[Interpolator]: 

44 """Retrieves the appropriate interpolation class based on the selected 

45 calculation type. 

46 

47 This method returns the corresponding interpolator class for assigning 

48 pixel values to FEM elements, depending on the current calculation 

49 type. Supported methods include interpolation at element nodes, element 

50 centers, or averaging all voxels within an element. 

51 

52 Returns: 

53 Type[Interpolator]: The interpolator class that matches the 

54 specified calculation type. 

55 """ 

56 

57 return { 

58 CalculationType.NODES: InterpolatorNodes, 

59 CalculationType.ALLVOXELS: InterpolatorAllVoxel, 

60 CalculationType.CENTER: InterpolatorCenter, 

61 }[self]

64def interpolate_image_to_discretization( 

65 dis: Discretization, image_data: ImageData, config: dict 

66) -> list[Element]: 

67 """Performs interpolation of image data onto the FEM Discretization based 

68 on the specified calculation type. 

69 

70 This function applies different interpolation methods depending on the 

71 user configuration. The pixel values are assigned to the FEM elements 

72 using one of the following approaches: 

73 

74 - "nodes": Computes the mean pixel value for each element based on its 

75 node values. 

76 - "allVoxel": Computes the mean pixel value for each element based on 

77 all voxels inside it. 

78 - "elementcenter": Assigns pixel values based on the center of each 

79 element. 

80 

81 Arguments: 

82 dis (Discretization): The Discretization object containing FEM 

83 elements and node coordinates. 

84 image_data (ImageData): A structured representation containing 3D 

85 pixel data, grid coordinates, orientation, and metadata. 

86 config (dict): User-defined configuration settings. 

87 

88 Returns: 

89 list[Element]: A list of FEM elements with interpolated pixel data. 

90 """ 

91 

92 calculation_type = CalculationType( 

93 config["processing options"]["calculation_type"] 

94 ) 

95 

96 interpolator = cast(Interpolator, calculation_type.get_interpolator()()) 

97 

98 return interpolator.compute_element_data(dis, image_data)
