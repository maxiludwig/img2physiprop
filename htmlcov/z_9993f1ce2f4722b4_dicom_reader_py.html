18class DicomReader(ImageReader): 

19 """A class for reading, processing, and organizing DICOM image data. 

20 

21 This class is designed to load DICOM image slices from a specified 

22 folder, filter and sort them based on spatial attributes, and 

23 convert them into structured data for 3D image reconstruction. It 

24 preserves critical metadata such as orientation, position, and 

25 pixeltype. 

26 """ 

27 

28 def _sort_dicoms( 

29 self, dicoms: list[FileDataset], slice_direction: np.ndarray 

30 ) -> list[FileDataset]: 

31 """Sorts a list of DICOM datasets in ascending order along a specified 

32 direction vector. Sorting is based on the projection of each slice's 

33 position onto the direction vector. 

34 

35 Arguments: 

36 dicoms (List[FileDataset]): A list of DICOM datasets, each 

37 containing the `ImagePositionPatient` attribute representing 

38 the slice's position in space. 

39 slice_direction (np.ndarray): A 3D vector defining the direction 

40 along which the slices should be sorted. 

41 

42 Returns: 

43 List[FileDataset]: The sorted list of DICOM datasets. 

44 """ 

45 

46 projections = [ 

47 np.dot(np.array(ds.ImagePositionPatient), slice_direction) 

48 for ds in dicoms 

49 ] 

50 

51 sorted_indices = np.argsort(projections) 

52 

53 sorted_dicoms = [dicoms[i] for i in sorted_indices] 

54 

55 return sorted_dicoms

57 def load_image(self, folder_path: Path) -> list[FileDataset]: 

58 """Loads and sorts DICOM image data from the specified folder. 

59 

60 This function searches for all `.dcm` files in the given folder, 

61 reads them using `pydicom`, and filters out any files that do not 

62 contain the `ImagePositionPatient` attribute. 

63 

64 Arguments: 

65 folder_path (Path): Path to the folder containing DICOM files. 

66 

67 Returns: 

68 list[FileDataset]: A list of DICOM datasets with valid 

69 image data, representing a 3D image when combined. 

70 """ 

71 

72 logging.info("Load image data!") 

73 

74 files = [] 

75 for fname in folder_path.glob("*.dcm"): 

76 files.append(pydicom.dcmread(fname)) 

77 

78 raw_dicoms = [] 

79 

80 skipcount = 0 

81 for f in files: 

82 if hasattr(f, "ImagePositionPatient"): 

83 raw_dicoms.append(f) 

84 else: 

85 skipcount += 1 

86 

87 return raw_dicoms

89 def convert_to_image_data( 

90 self, raw_dicoms: list[FileDataset] 

91 ) -> ImageData: 

92 """Processes and structures DICOM datasets into a 3D image 

93 representation. 

94 

95 This function extracts pixel data and metadata from a list of DICOM 

96 slices, sorts them along the slice direction, and filters slices 

97 within a specified bounding box. It constructs a structured 3D image 

98 with pixel data, grid coordinates, orientation, position, and pixel 

99 type. 

100 

101 Arguments: 

102 raw_dicoms (list[FileDataset]): A list of DICOM datasets 

103 representing 2D slices of a 3D image. 

104 

105 Returns: 

106 ImageData: A structured representation of the 3D image, including 

107 pixel data, grid coordinates, orientation, and metadata. 

108 

109 

110 Raises: 

111 RuntimeError: If the DICOM modality is not supported. 

112 RuntimeError: If Dicom is not in bounding box. 

113 

114 Notes: 

115 - The function assumes all slices share the same orientation, 

116 pixel spacing, and slice thickness. 

117 """ 

118 

119 try: 

120 pixel_type = PixelValueType(raw_dicoms[0].Modality) 

121 

122 except ValueError: 

123 raise RuntimeError("Modality not supported") 

124 

125 row_direction = np.array(raw_dicoms[0].ImageOrientationPatient[3:]) 

126 column_direction = np.array(raw_dicoms[0].ImageOrientationPatient[:3]) 

127 slice_direction = np.cross(column_direction, row_direction) 

128 

129 slice_orientation = self._get_slice_orientation( 

130 row_direction, column_direction 

131 ) 

132 

133 sorted_dicoms = self._sort_dicoms(raw_dicoms, slice_direction) 

134 

135 raw_pixel_data = [] 

136 coords_in_crop = [] 

137 

138 for dicom in tqdm(sorted_dicoms, desc="Processing Elements"): 

139 

140 if slice_orientation.is_within_crop( 

141 dicom.ImagePositionPatient, self.bounding_box 

142 ): 

143 

144 raw_pixel_data.append(dicom.pixel_array) 

145 coords_in_crop.append(dicom.ImagePositionPatient) 

146 

147 else: 

148 continue 

149 

150 if not raw_pixel_data: 

151 raise RuntimeError( 

152 "No slice images found within the volume of the imported mesh." 

153 ) 

154 

155 slope = float(getattr(sorted_dicoms[0], "RescaleSlope", 1.0)) 

156 intercept = float(getattr(sorted_dicoms[0], "RescaleIntercept", 0.0)) 

157 pixel_data = np.array(raw_pixel_data) * slope + intercept 

158 

159 slice_spacing = float( 

160 getattr( 

161 sorted_dicoms[0], 

162 "SpacingBetweenSlices", 

163 sorted_dicoms[0].SliceThickness, 

164 ) 

165 ) 

166 

167 N_slice, N_col, N_row = pixel_data.shape 

168 

169 slice_coords = np.arange(N_slice) * slice_spacing 

170 row_coords = np.arange(N_row) * sorted_dicoms[0].PixelSpacing[0] 

171 col_coords = np.arange(N_col) * sorted_dicoms[0].PixelSpacing[1] 

172 

173 return ImageData( 

174 pixel_data=np.array(pixel_data), 

175 grid_coords=GridCoords(slice_coords, row_coords, col_coords), 

176 orientation=np.column_stack( 

177 (slice_direction, row_direction, column_direction) 

178 ), 

179 position=coords_in_crop[0], 

180 pixel_type=pixel_type, 

181 )
